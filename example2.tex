int MAD() {
 sum:=$\cur$;cnt:=0;$\next$;
 loop{
  sum+=$\cur$;cnt+=1;$\next$;
 };
 $\init$;avg:= sum/cnt;mad:=0;
 loop{
  if($\cur$<avg){
   mad=mad+(avg-$\cur$);
  }else{
   mad=mad+($\cur$-avg);
  };$\next$;
 }
 return mad/cnt;
}
\end{lstlisting}

 sum+=$\cur$;cnt+=1;$\next$;
 };
 init(); avg:= sum/cnt;sd:=0;
 loop{
  sd+=($\cur$-avg)^2;next();
 }
 return SQRT(sd/cnt);
}
