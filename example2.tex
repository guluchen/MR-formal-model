int MAD() {
 sum:=$\cur$;cnt:=0;$\nnext$;
 loop{
  sum+=$\cur$;cnt+=1;$\nnext$;
 };
 avg:= sum/cnt;mad:=0;$\init$;
 loop{
  if($\cur$<avg){
   mad=mad+(avg-$\cur$);
  }else{
   mad=mad+($\cur$-avg);
  };$\nnext$;
 }
 return mad/cnt;
}
\end{lstlisting}

 sum+=$\cur$;cnt+=1;$\nnext$;

 avg:= sum/cnt;sd:=0;init();
 loop{
  sd+=($\cur$-avg)*($\cur$-avg);next();
 }
 return SQRT(sd/cnt);
}
